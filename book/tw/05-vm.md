# 第 5 章：虛擬機器

## 5.1 位元組碼與虛擬機器

位元組碼 (Bytecode) 是程式中間表示的一種，類似於機器碼但由軟體虛�擬機器執行。我們的虛擬機器是**堆疊式虛擬機器 (Stack-based VM)**。

```
AST                           位元組碼                      堆疊
────────                     ──────────                   ────
10 + 20 * 3                  OP_CONST 10                   [10]
                             OP_CONST 20                   [10, 20]
                             OP_CONST 3                    [10, 20, 3]
                             OP_MULT                        [10, 60]
                             OP_ADD                         [70]
```

## 5.3 虛擬機器結構

[程式檔案：05-1-vm.c (part 1)](../_code/05/05-1-vm.c)
```c
#include "opcodes.h"
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define STACK_MAX 256
#define GLOBAL_MAX 256

typedef struct {
    OpCode opcode;
    int operand;
} Instruction;

typedef struct {
    Instruction* code;
    int count;
    int capacity;
} Chunk;

typedef Value Stack[STACK_MAX];
typedef Value Globals[GLOBAL_MAX];

typedef struct VM {
    Chunk* chunk;
    int ip;
    Stack stack;
    int sp;
    Globals globals;
    int globalCount;
} VM;

static VM vm;

void initVM() {
    vm.chunk = NULL;
    vm.ip = 0;
    vm.sp = 0;
    vm.globalCount = 0;
}

void freeVM() {
    if (vm.chunk) free(vm.chunk);
}
```

## 5.4 堆疊操作

```c
static void push(Value value) {
    if (vm.sp >= STACK_MAX) {
        fprintf(stderr, "Stack overflow!\n");
        exit(1);
    }
    vm.stack[vm.sp++] = value;
}

static Value pop() {
    if (vm.sp <= 0) {
        fprintf(stderr, "Stack underflow!\n");
        exit(1);
    }
    return vm.stack[--vm.sp];
}

static Value peek() {
    if (vm.sp <= 0) {
        fprintf(stderr, "Stack underflow!\n");
        exit(1);
    }
    return vm.stack[vm.sp - 1];
}
```

## 5.5 位元組碼生成

```c
void writeChunk(Chunk* chunk, OpCode opcode, int operand) {
    if (chunk->count >= chunk->capacity) {
        chunk->capacity = chunk->capacity == 0 ? 8 : chunk->capacity * 2;
        chunk->code = realloc(chunk->code, sizeof(Instruction) * chunk->capacity);
    }
    chunk->code[chunk->count].opcode = opcode;
    chunk->code[chunk->count].operand = operand;
    chunk->count++;
}

#define EMIT_BYTE(vm, opcode) writeChunk(vm.chunk, opcode, 0)

static int addConstant(Chunk* chunk, Value value) {
    EMIT_BYTE(chunk, OP_CONSTANT);
    int index = chunk->count;
    EMIT_BYTE(chunk, (OpCode)value.as.number);
    return index;
}
```

## 5.6 指令分派迴圈

[程式檔案：05-1-vm.c (part 2)](../_code/05/05-1-vm.c)
```c
static InterpretResult run() {
    for (;;) {
#ifdef DEBUG_TRACE_EXECUTION
        printf("          ");
        for (int i = 0; i < vm.sp; i++) {
            printf("[ ");
            printValue(vm.stack[i]);
            printf(" ]");
        }
        printf("\n");
        disassembleInstruction(vm.chunk, vm.ip);
#endif

        OpCode instruction = vm.chunk->code[vm.ip++].opcode;

        switch (instruction) {
            case OP_CONSTANT: {
                Value constant = { VAL_NUMBER };
                constant.as.number = vm.chunk->code[vm.ip++].operand;
                push(constant);
                break;
            }

            case OP_NIL:
                push(NIL_VAL);
                break;

            case OP_TRUE:
                push(BOOL_VAL(true));
                break;

            case OP_FALSE:
                push(BOOL_VAL(false));
                break;

            case OP_ADD: {
                Value b = pop();
                Value a = pop();
                Value result = { VAL_NUMBER };
                result.as.number = a.as.number + b.as.number;
                push(result);
                break;
            }

            case OP_SUBTRACT: {
                Value b = pop();
                Value a = pop();
                Value result = { VAL_NUMBER };
                result.as.number = a.as.number - b.as.number;
                push(result);
                break;
            }

            case OP_MULTIPLY: {
                Value b = pop();
                Value a = pop();
                Value result = { VAL_NUMBER };
                result.as.number = a.as.number * b.as.number;
                push(result);
                break;
            }

            case OP_DIVIDE: {
                Value b = pop();
                Value a = pop();
                Value result = { VAL_NUMBER };
                result.as.number = a.as.number / b.as.number;
                push(result);
                break;
            }

            case OP_NEGATE:
                push(NEGATE(pop()));
                break;

            case OP_RETURN: {
                printValue(pop());
                printf("\n");
                return INTERPRET_OK;
            }
        }
    }
}
```

## 5.7 執行器入口

```c
typedef enum {
    INTERPRET_OK,
    INTERPRET_COMPILE_ERROR,
    INTERPRET_RUNTIME_ERROR
} InterpretResult;

InterpretResult interpret(Chunk* chunk) {
    vm.chunk = chunk;
    vm.ip = 0;
    vm.sp = 0;
    return run();
}
```

## 5.9 除錯支援

```c
void disassembleChunk(Chunk* chunk, const char* name) {
    printf("=== %s ===\n", name);
    for (int offset = 0; offset < chunk->count;) {
        offset = disassembleInstruction(chunk, offset);
    }
}

int disassembleInstruction(Chunk* chunk, int offset) {
    printf("%04d ", offset);
    OpCode instruction = chunk->code[offset].opcode;

    switch (instruction) {
        case OP_CONSTANT:
            printf("OP_CONSTANT %d\n", chunk->code[offset + 1].operand);
            return offset + 2;
        case OP_ADD:
            printf("OP_ADD\n");
            return offset + 1;
        case OP_RETURN:
            printf("OP_RETURN\n");
            return offset + 1;
        default:
            printf("Unknown opcode %d\n", instruction);
            return offset + 1;
    }
}
```
